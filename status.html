async function verificaStatus() { const cod = document.getElementById('order-id-input').value.trim().toUpperCase(); const box = document.getElementById('status-box'); const titlu = document.getElementById('status-title'); const desc = document.getElementById('status-desc'); if (!cod) return alert("Introdu un cod!"); try { // Adăugăm un număr aleatoriu la finalul link-ului (?nocache=...) ca să păcălim browserul să nu folosească date vechi let res = await fetch(`https://api.jsonbin.io/v3/b/${BIN_ID}/latest?nocache=${Date.now()}`, { headers: { "X-Master-Key": API_KEY } }); let json = await res.json(); let comenzi = json.record.statusuriComenzi; if (comenzi && comenzi[cod]) { titlu.innerText = "Status: " + comenzi[cod].status; desc.innerText = comenzi[cod].descriere || "Fără detalii suplimentare."; box.style.display = "block"; } else { alert("Codul de comandă nu a fost găsit! Verifică dacă l-ai scris corect."); box.style.display = "none"; } } catch (e) { alert("Eroare la conectarea cu serverul!"); box.style.display = "none"; } }
